<script>
    function readURL(input, img) {
        if (input.files && input.files[0]) {
            var reader = new FileReader();

            reader.onload = function (e) {
                $(img).attr('src', e.target.result);
            };

            reader.onloadend = function () {
                $(img).attr('src', reader.result);
            };

            reader.readAsDataURL(input.files[0]);
        }
    }

    function getFilesize(fileInput) {
        var sizeinbytes;
        if (window.ActiveXObject) {
            var fso = new ActiveXObject("Scripting.FileSystemObject");
            var filepath = fileInput.value;
            var thefile = fso.getFile(filepath);
            sizeinbytes = thefile.size;
        } else {
            if (fileInput.files.length > 0) {
                sizeinbytes = fileInput.files[0].size;
            }
        }

        return sizeinbytes;
    }


    function resetFormElement(e) {
        e.wrap('<form>').closest('form').get(0).reset();
        e.unwrap();
        if (e.parent().find("img").length > 0) {
            var img = e.parent().find("img").first();
            img.remove();
        }
    }


    function checkImageType(fileTypeStr, thisFile) {
        var class_attr = thisFile.attr("class")
        if (!(class_attr.includes("extra_file"))) {

            if ($.inArray(fileTypeStr, ['JPG', 'JPEG', 'jpg', 'jpeg']) == -1) {

                var attr = thisFile.attr('id');
                var errorSpan = document.getElementById(attr + "typeError");
                if (errorSpan == null) {
                    var x = document.createElement("SPAN");
                    var t = document.createTextNode("پسوند فایل بارگذاری شده مورد قبول نمی باشد. لطفا فقط از فایل با فرمت jpg یا jpeg استفاده نمایید.");
                    x.setAttribute("class", "help-block m-b-none");
                    x.setAttribute("style", "color: #ff0000");
                    x.setAttribute("id", attr + "typeError");
                    x.appendChild(t);
                    var selected_div = thisFile.closest('div[class="col-sm-7"]');
                    selected_div.append(x);
                }
                return 1;


            } else {
                return 0;
            }
        } else {
            return 0;
        }

    }

    function validate_image(file, show = true) {
        var ext = $(file).val().split('.').pop().toLowerCase();
        var flag = checkImageType(ext, $(file));
        var size = getFilesize(file);

        //  if (size > GLOBAL_MAX_FILE_SIZE || flag) {
        //$(file).parents(".form-group").addClass("has-error");
        //
        //    resetFormElement($(file));
        //      return;
        //    }
        var selector = $(file).attr('name') + "_cache_key";

        var img;
        if ($(file).parent().find("img").length > 0) {
            img = $(file).parent().find("img").first();
        } else {
            img = $("<img class='preview-img ibox-content text-center' />");
            $(file).after(img);
        }
        if (flag) {
            $(file).val("")
        }
        if (show) {
            readURL(file, img);
            $(img).height(200);
        }
    }

    $(document).ready(function () {
        $("form img[src='/static/images/base/default-user.png']").each(function () {
            $(this).parents(".form-group").addClass("has-error");
        });

        $("input[type=file]").change(function () {
            var ext = $(this).val().split('.').pop().toLowerCase();
            var flag = checkImageType(ext, $(this));
            var size = getFilesize(this);
            if (size > GLOBAL_MAX_FILE_SIZE || flag) {
                $(this).parents(".form-group").addClass("has-error");
                resetFormElement($(this));
                return;
            } else {
                $(this).parents(".form-group").removeClass('has-error');
            }
            var img;
            if ($(this).parent().find("img").length > 0) {
                img = $(this).parent().find("img").first();
            } else {
                img = $("<img class='preview-img ibox-content text-center' />");
                $(this).after(img);
            }
            has_error = $(this).parents(".form-group").hasClass("has-error");
            is_extra = $(this).attr("id").includes("extra")
            if (!has_error && is_extra){
                readURL(this, img);
                $(img).height(200);
            }

        });
    });

                form.validate().settings.ignore = ":disabled";
                var hasError = hasIssueDateErr || hasExpireDateErr || hasBirthDateErr || hasPassportNumberErr;
                if (hasError) {
                    return false;

            function verify_modal(form, name) {
                $('#register_form').validate({
                    errorPlacement: function (error, element) {
                        element.before(error);
                    },
                    rules: {
                        confirm: {
                            equalTo: "#password"
                        },
                        phone_number: {
                            digits: true
                        },
                        mobile: {
                            digits: true
                        },
                        iran_phone_number: {
                            digits: true
                        },
                        email: {
                            email: true
                        },
                        birth_date: {
                            date: true
                        },
                        expire_date: {
                            date: true
                        },
                        passport_issue_date: {
                            date: true
                        }
                    }
                });
                var visa_type;
                var nationality;
                var entry_option;
                var issuer_agent;
                var passport_type;
                var is_urgent;
                var float;
                var country_extrafield;

                if ($('#register_form').valid()) {
                    issuer_agent = $("input[name='issuer_agent_id']").val();
                    visa_type = $("select[name='visa_type']").children("option:selected").val();
                    nationality = $("select[name='nationality']").children("option:selected").val();
                    entry_option = $("select[name='entry_option']").children("option:selected").val();
                    country_extrafield = $('#country_extrafield_1').val();
                    passport_type = $("select[name='passport_type']").children("option:selected").val();
                    is_urgent = $("input[name='is_urgent']").is(':checked');
                    var lang = "fa";
                    float = "right"
                    if (lang == 'ar' || 'fa'){
                        float = "left"
                    }

                    $('#verifyModal .modal-body').html("");
                    var fields = $(form).find('div.form-group');
                    var formFields = new Array();
                    var table = $("<table classs='table'>");
                    var div = $("<div class='placeOfIssue'>");

                    for (i = 0; i < fields.length; i++) {
                        //Check if in this form-group input type if not file type
                        if (($(fields[i]).css('display') == 'none' || ($(fields[i]).find('img').length > 0))) {
                            continue;
                        }
                        if ($(fields[i]).find('input[type=file]').length <= 0) {
                            var labelTxt = $(fields[i]).find('label.control-label').html();

                            var spanIndex = labelTxt.indexOf('<');

                            if (spanIndex != -1) {
                                labelTxt = labelTxt.substring(0, spanIndex)
                            }

                            var inputTxt = $(fields[i]).find('input[type=text]').val()
                                || $(fields[i]).find('input[type=email]').val()
                                || $(fields[i]).find('textarea').val()
                                || $(fields[i]).find('input[type=checkbox]').is(':checked')
                                || $($(fields[i]).find('span.select2-chosen')[0]).text();


                            formFields.push(
                                {
                                    label: labelTxt,
                                    input: inputTxt
                                });
                            if (inputTxt == true) {
                                inputTxt = "بله"
                            } else if (inputTxt == false) {
                                inputTxt = "خیر"
                            }
                            var emailclass = ''
                            if (inputTxt.includes("@")) {
                                emailclass = 'emailInput'
                            }
                            if (i % 2 == 0) {
                                table.append("<tr><td style='font-size:15px;'class='" + emailclass + "'>" + labelTxt + "</td><td style='font-weight:700;' class='" + emailclass + "'>" + inputTxt + "</td>");
                            } else {
                                table.append("<tr class='highlighted'><td style='font-size:15px' class='" + emailclass + "'>" + labelTxt + "</td><td style='font-weight:700;' class='" + emailclass + "'>" + inputTxt + "</td>");
                            }

                        }
                    }

                    div.append("<img src='/static/theme/img/issue.png' width='100' style='margin: auto 5px;float: " + float + "'><div><span style='font-size:15px;color:#666a6c;'>" + $('#id_value_issuer_agent').parent().parent().find('label.control-label').clone().children().remove().end().text() + "</span><span style='font-weight:700;color:#666a6c;    width: 50%;margin: 0px 20px;display: inline-table;'>" + $('#id_value_issuer_agent').val() + "</span><br>");
                    $.ajax({
                    type: "POST",
                    url: "/fa/request/get_visa_fee/",
                    data: {
                        'visa_type': visa_type,
                        'nationality': nationality,
                        'issuer':issuer_agent,
                        'entry_option':entry_option,
                        'is_urgent':is_urgent,
                        'passport_type':passport_type,
                        'country_extrafield': country_extrafield,

                    },
                    dataType: "json",
                    success: function (response) {
                        if ( parseInt(response['visa_fee']) == 0){
                            div.append("<div class='visa_free' id='visa_fee_display'><span style='font-size:15px;color:#666a6c;'>هزینه روادید</span><span style='font-weight:700;color:#41b394;    width: 40%;margin: 10px 41px 10px 61px;display: inline-table;'>رایگان</span></div>");
                        }
                        if ( parseInt(response['visa_fee']) > 0){
                            div.append("<div class='visa_free' id='visa_fee_display'><span style='font-size:15px;color:#666a6c;'>هزینه روادید</span><span style='font-weight:700;color:#41b394;    width: 40%;margin: 10px 41px 10px 61px;display: inline-table;'>" + response['visa_fee'] + "&nbsp" + response['currency_unit'] + "</span></div>");
                        }
                            div.append("<div><span style='font-size:15px;color:#666a6c;text-align: center;float: right;margin: 10px 0px 20px;background: rgba(0,0,0,0.03);padding: 6px;'>اعتبار روادید انتخابی شما از زمان صدور حداکثر به مدت <span style='font-weight:700;'>" + response['max_validity'] + "</span> روز می باشد. همچنین مدت اقامت شما در ایران نباید بیشتر از <span style='font-weight:700;'>" + $('#id_duration').val() + "</span> روز باشد. </span></div></div>");




                    },
                    error: function (response) {
                        if (response.status == 403) {
                            $("#visa_fee_display").html("");
                            var $html = "<div class='alert alert-danger'>" +
                                        "<p><b>ارسال بیش از حد مجاز</b></p>" +
                                        "<p>سامانه روادید بیش از حد مجاز از سوی دستگاه شما درخواست دریافت کرده است. <br> لطفا تا لحظاتی دیگر مجددا تلاش نمایید. </p></div>";
                            $('#id_box_passport').append($html);
                        }
                        if (response.status == 500){
                            var visa_block = $("#visa_fee_display")
                            if (visa_block.length)
                            $("#visa_fee_display").hide();

                        }

                    }
                });

                    $('#verifyModal .modal-body').append(table);
                    $('#verifyModal .modal-body').append(div);
                    $('#verifyModal').modal();
                }
                else {
                    $('.error').first().next().focus();

                    swal({
                        title: "در فرم تقاضای روادید خطا وجود دارد. لطفا با دقت فرم را تکمیل نمایید.",
                        type: "error",
                        confirmButtonText: "قبول",
                        showCancelButton: false,
                        closeOnConfirm: true,
                        animation: "slide-from-top"
                    }, function () {
                        $('html, body').animate({scrollTop: $('.error').first().next().offset().top - 120}, 'slow');

                    });
                }
            }
